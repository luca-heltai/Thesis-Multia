Given an arbitrary field $\gamma$, weâ€™ll use the following shorthand notation:

\begin{equation}
  \label{eq:2}
  \begin{aligned}
    \vv A^\gamma_{ij} & = (\gamma N_j', N_i') &  = \int_0^1 \gamma(s) N_j'(s)
    N_i'(s) \d s \\
    \vv B^\gamma_{ij} & = (\gamma N_j', N_i) & = \int_0^1 \gamma(s) N_j'(s)
    N_i(s) \d s \\
    \vv M^\gamma_{ij} & = (\gamma N_j, N_i) & = \int_0^1 \gamma(s) N_j(s)
    N_i(s) \d s 
  \end{aligned}
\end{equation}
